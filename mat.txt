
// Função somar
int somar(int a, int b) {
  int s = 0;
  s = a + b;
  return s;
}

// Função subtrair
int subtrair(int a, int b) {
  int s = 0;
  s = a - b;
  return s;
}

// Função multiplicação
int multiplicacao(int a, int b) {
  int s = 0;
  s = a * b;
  return s;
}

// Função divisão
double divisao(int a, int b) {
  double s = 0;
  s = a / b;
  return s;
}
